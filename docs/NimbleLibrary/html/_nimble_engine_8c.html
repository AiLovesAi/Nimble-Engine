Allocates a pointer.

Allocates a pointer and checks if it is successful.

Example:

#include <stdio.h>
#include <stdlib.h>
#include <Nimble/NimbleEngine.h>

{
char original[] = "Hello world.";
char *new = nAlloc(sizeof(original) + 1);
new[sizeof(original) - 1] = '\n';
new[sizeof(original)] = '\0';
puts(new);
new = nFree(new);
if (new)
{
puts("We will get here only if we use nFree(new) with no assignment!");
}
return EXIT_SUCCESS;
}

Initializes the game engine by setting up the error/crash handlers, logger, and thread manager.

Example:
#include <inttypes.h>
#include <Nimble/NimbleEngine.h>
int main(int argc, char **argv)
{
if (nEngineInit() != NSUCCESS)
{
fprintf(stderr, "Failed to initialize Nimble.");
exit(EXIT_FAILURE);
}
printf("Successfully initialized game engine.");
exit(EXIT_SUCCESS);
}

char* nStringCopy ( char *  dst,
const char *  src,
const size_t  len 
)

Copies len characters from src to dst.

Copies len characters from src to dst. The string is always null terminated. If there is already a null terminator, no more characters are copied.

Example:

#include <stdio.h>
#include <stdlib.h>
#include <Nimble/NimbleEngine.h>
int main(int argc, char **argv)
{
char original[] = "Hello world.";
char *new = nAlloc(sizeof(original));
nStringCopy(original, new, sizeof(original) - 1);
printf("New string: %s\n", new);
return EXIT_SUCCESS;
}
Parameters
[out]dstThe pointer to the destination.
[in]srcThe pointer to free.
[in]lenThe number of characters to copy. This must not include the null terminator.
Returns
dst is always returned.
